\usetikzlibrary{decorations.text}
\newcommand\PathWithText[1]{
\draw (0,0) circle (1cm);
\path[
  postaction={
    decorate,
      decoration={
        text along path,
        reverse path=false,
        text={#1}
      }
    }
] (0,0) circle (1cm);
}
